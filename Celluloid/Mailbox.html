Celluloid::Mailbox

Actors communicate with asynchronous messages. Messages are buffered in Mailboxes until Actors can act upon them.

def <<(message)
  @lock.synchronize do
    raise MailboxError, "dead recipient" if @dead
    
    @messages << message
    @condition.signal
  end
  nil
end

Receive a message from the Mailbox

# File lib/celluloid/mailbox.rb, line 41
def receive
  message = nil
  
  @lock.synchronize do
    raise MailboxError, "attempted to receive from a dead mailbox" if @dead
    
    begin
      if block_given?
        index = @messages.index do |msg|
          yield(msg) || msg.is_a?(Celluloid::SystemEvent)
        end

        message = @messages.slice!(index, 1).first if index
      else
        message = @messages.shift
      end
      
      raise message if message.is_a?(Celluloid::SystemEvent)
      
      @condition.wait(@lock) unless message
    end until message
  end
    
  message
end

def shutdown
  @lock.synchronize do
    @messages.each { |msg| msg.cleanup if msg.respond_to? :cleanup }
    @messages.clear
    @dead = true
  end
end

def system_event(event)
  @lock.synchronize do
    unless @dead # Silently fail if messages are sent to dead actors
      @messages.unshift event
      @condition.signal
    end
  end
  nil
end
