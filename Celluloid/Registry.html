def []=(name, actor)
  actor_singleton = class << actor; self; end
  unless actor_singleton.ancestors.include? ActorProxy
    raise TypeError, "not an actor"
  end

  @@registry_lock.synchronize do
    @@registry[name.to_sym] = actor
  end
end
