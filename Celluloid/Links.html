# File lib/celluloid/linking.rb, line 50
def inspect
  @lock.synchronize do
    links = @links.to_a.map { |l| "#{l.class}:#{l.object_id}" }.join(',')
    "#<Celluloid::Links[#{links}]>"
  end
end
