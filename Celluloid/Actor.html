Overview

Actors are Celluloid’s concurrency primitive. They’re implemented as normal Ruby objects wrapped in threads which communicate with asynchronous messages.

def initialize(subject)
  @subject  = subject
  @mailbox = initialize_mailbox
  @links   = Links.new
  @signals = Signals.new
  @proxy   = ActorProxy.new(self, @mailbox)
  @running = true

  @thread = Pool.get
  @thread[:queue] << proc do
    initialize_thread_locals
    run
  end
end

def initialize_thread_locals
  Thread.current[:actor]       = self
  Thread.current[:actor_proxy] = @proxy
  Thread.current[:mailbox]     = @mailbox
end

def run
  process_messages
  cleanup ExitEvent.new(@proxy)
rescue Exception => ex
  @running = false
  handle_crash(ex)
ensure
  Pool.put @thread
end
