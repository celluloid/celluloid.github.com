def create
  queue = Queue.new
  thread = Thread.new do
    begin
      while true
        queue.pop.call
      end
    rescue Exception => ex
      # Rubinius hax
      raise if defined?(Thread::Die) and ex.is_a? Thread::Die

      message = "Celluloid::Actor::Pool internal failure:\n"
      message << "#{ex.class}: #{ex.to_s}\n"
      message << ex.backtrace.join("\n")
      Celluloid.logger.error message if Celluloid.logger
    end
  end
  thread[:queue] = queue
  thread
end

def get
  @lock.synchronize do
    if @pool.empty?
      create
    else
      @pool.shift
    end
  end
end

def put(thread)
  @lock.synchronize do
    if @pool.size >= @max_idle
      thread.kill
    else
      @pool << thread
    end
  end
end
