# File lib/celluloid/signals.rb, line 17
def send(name, value = nil)
  fibers = @waiting.delete name
  return unless fibers
  
  fibers.each { |fiber| fiber.resume value }
  true
end

def wait(name)
  fibers = @waiting[name] ||= []
  fibers << Fiber.current
  Fiber.yield
end
