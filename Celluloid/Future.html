def initialize(*args, &block)
  @lock = Mutex.new
  @value_obtained = false
  
  @runner = Runner.new(*args, &block)
  @runner.run!
end

def value
  @lock.synchronize do
    unless @value_obtained
      @value = @runner.value
      @runner.terminate
      @value_obtained = true
    end
    
    @value
  end
end
