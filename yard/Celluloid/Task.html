def initialize(type)
  @type   = type
  @status = :new

  actor   = Thread.current[:actor]
  mailbox = Thread.current[:mailbox]

  @fiber = Fiber.new do
    @status = :running
    Thread.current[:actor]   = actor
    Thread.current[:mailbox] = mailbox
    Fiber.current.task = self
    actor.tasks << self

    begin
      yield
    rescue TerminatedError
      # Task was explicitly terminated
    ensure
      actor.tasks.delete self
    end
  end
end

def self.suspend(status)
  task = Task.current
  task.status = status

  result = Fiber.yield
  raise TerminatedError, "task was terminated" if result == TerminatedError
  task.status = :running

  result
end

def resume(value = nil)
  @fiber.resume value
  nil
rescue FiberError
  raise DeadTaskError, "cannot resume a dead task"
rescue RuntimeError => ex
  # These occur spuriously on 1.9.3 if we shut down an actor with running tasks
  return if ex.message == ""
  raise
end
