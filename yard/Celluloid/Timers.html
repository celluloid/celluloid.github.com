def fire
  return if @timers.empty?

  time = Time.now + Timer::QUANTUM
  while not empty? and time > @timers.first.time
    timer = @timers.shift
    timer.call
  end
end

def index(timer)
  l, r = 0, @timers.size - 1

  while l <= r
    m = (r + l) / 2
    if timer < @timers.at(m)
      r = m - 1
    else
      l = m + 1
    end
  end
  l
end

def wait
  return if @timers.empty?

  interval = wait_interval
  sleep interval if interval >= Timer::QUANTUM
  fire
end
